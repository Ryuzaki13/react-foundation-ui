## Основной сценарий

<Primary />

<Controls of={ODataSelectStories.BasicDivision} />

```tsx
import { useState } from "react";

import { ODataSelect } from "@/shared/ui";

export function Example() {
	const [value, setValue] = useState<string | undefined>(undefined);

	return (
		<ODataSelect
			label="Регион"
			value={value}
			onChange={setValue}
			odata={{ service: "DEMO_REFERENCE_SRV", target: "DemoReferenceItems" }}
			model={{ codeKey: "REGION" }}
			segment={{ placeholder: "Регион" }}
		/>
	);
}
```
