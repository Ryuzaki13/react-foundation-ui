## Основной сценарий

<Primary />

<Controls of={SelectStories.ControlledObjects} />

```tsx
import { useState } from "react";

import { Select } from "@/shared/ui";

const options = [
	{ id: 1, name: "Отдел продаж", code: "SLS" },
	{ id: 2, name: "Закупки", code: "PRC" }
];

export function Example() {
	const [value, setValue] = useState(undefined);

	return (
		<Select
			label="Подразделение"
			value={value}
			onChange={setValue}
			options={options}
			getOptionKey={(option) => option.id}
			getOptionLabel={(option) => option.name}
			getOptionCode={(option) => option.code}
		/>
	);
}
```
