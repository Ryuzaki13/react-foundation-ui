```tsx
import { useState } from "react";

import { RadioGroup } from "@/shared/ui";

export function Example() {
	const [value, setValue] = useState("email");

	return (
		<RadioGroup label="Канал" value={value} onChange={setValue}>
			<RadioGroup.Option value="email" label="Email" />
			<RadioGroup.Option value="sms" label="SMS" />
		</RadioGroup>
	);
}
```
