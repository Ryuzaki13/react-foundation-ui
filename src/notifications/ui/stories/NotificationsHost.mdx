```tsx
import { Button } from "@/shared/ui";
import { NotificationApi, NotificationsHost, NotificationsProvider } from "@/shared/ui";

export function Example() {
	return (
		<NotificationsProvider>
			<Button
				onClick={() =>
					NotificationApi.push({
						title: "Готово",
						description: "Операция завершена",
						tone: "success"
					})
				}>
				Показать уведомление
			</Button>
			<NotificationsHost />
		</NotificationsProvider>
	);
}
```
