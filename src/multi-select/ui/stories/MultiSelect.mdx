## Основной сценарий

<Primary />

<Controls of={MultiSelectStories.Basic} />

```tsx
import { useState } from "react";

import { MultiSelect } from "@/shared/ui";

const items = [
	{ code: "MOW", label: "Москва" },
	{ code: "EKB", label: "Екатеринбург" }
];

export function Example() {
	const [value, setValue] = useState([]);
	const [query, setQuery] = useState("");

	return (
		<MultiSelect
			label="Города"
			items={items}
			value={value}
			query={query}
			onQuery={setQuery}
			onChange={setValue}
			codeKey="code"
			textKey="label"
		/>
	);
}
```
